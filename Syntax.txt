CREATE TABLE <TABLE NAME>(
<COLUMN NAME> <DATA TYPE><(SIZE)> <KEY>,
<COLUMN NAME> <DATA TYPE><(SIZE)>,
<COLUMN NAME> <DATA TYPE><(crSIZE)>);

NOTE: No spaces in column names are allowed.

2.DROP:
	DROP TABLE <TABLE NAME>;

3.SELECT:
	SELECT * FROM <TABLE NAME>;

	{here * means all}

4.INSERT:
	INSERT INTO <TABLE NAME> VALUES(<COLUMN1 VALUE>, <COLUMN2 VALUE>, .....,<COLUMN-N VALUE>);

5.UPDATE:
	UPDATE <TABLE NAME> SET <COLUMN NAME> = <NEW VALUE>
	WHERE <COLUMN NAME> = <TARGET VALUE>;

6.DELETE:
	DELETE FROM <TABLE NAME>
	WHERE <COLUMN NAME> = <VALUE>;

NOTE: 'WHERE' clause.
